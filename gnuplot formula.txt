" 基础设置
set number                          " 显示行号
set autoindent                      " 启用自动缩进
set smartindent                     " 启用智能缩进
set tabstop=4                       " 制表符占用空格数
set shiftwidth=4                    " 缩进时的空格数
set expandtab                       " 将制表符转换为相应的空格数
set ignorecase                      " 搜索时忽略大小写
set smartcase                       " 如果搜索包含大写字符，则区分大小写
set wrap                            " 开启自动换行
set linebreak                       " 根据单词换行，避免单词被拆分
set showmatch                       " 显示匹配的括号
set encoding=utf-8                  " 设置文件编码
set mouse=a                         " 启用鼠标支持
syntax on                           " 启用语法高亮
filetype plugin indent on           " 启用文件类型检测

" 搜索相关设置

set incsearch                       " 输入搜索内容时即时显示搜索结果

"高量显示光标所在的行
set cursorline

"显示当前光标的位置信息
set ruler

autocmd InsertLeave,TextChanged * if !exists('g:autosaving') || g:autosaving | silent! wa | endif
